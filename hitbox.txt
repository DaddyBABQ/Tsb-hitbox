local function updateHistory(name, pos)
    if not posHistory[name] then posHistory[name]={} timeHistory[name]={} end
    table.insert(posHistory[name], pos)
    table.insert(timeHistory[name], tick())
    if #posHistory[name] > HISTORY then
        table.remove(posHistory[name],1) table.remove(timeHistory[name],1)
    end
end

local function getVelAndAccel(name)
    local pos = posHistory[name]
    local times = timeHistory[name]
    if not pos or #pos < 3 then return Vector3.zero, Vector3.zero end
    local totalVel, totalWeight = Vector3.zero, 0
    for i = 2, #pos do
        local dt = times[i]-times[i-1]
        if dt > 0 and dt < 0.15 then
            local w = i*i
            totalVel += ((pos[i]-pos[i-1])/dt)*w
            totalWeight += w
        end
    end
    local vel = totalWeight > 0 and (totalVel/totalWeight) or Vector3.zero
    local accel = Vector3.zero
    if #pos >= 3 then
        local dt1=times[#pos-1]-times[#pos-2]
        local dt2=times[#pos]-times[#pos-1]
        if dt1>0 and dt2>0 then
            local v1=(pos[#pos-1]-pos[#pos-2])/dt1
            local v2=(pos[#pos]-pos[#pos-1])/dt2
            accel=(v2-v1)/dt2
        end
    end
    return vel, accel
end

local function isSafeToRestore(oldCFrame)
    if oldCFrame.Position.Y < 400 then return false end
    if tick() < damagedUntil then return false end
    if math.abs(hrp.AssemblyLinearVelocity.Y) > 30 then return false end
    return true
end

local function findTarget()
    if getgenv().lockedTarget then
        local t = getgenv().lockedTarget
        if t and t.Parent then
            local h = t.Parent:FindFirstChildOfClass("Humanoid")
            if h and h.Health > 0 then return t end
        end
        getgenv().lockedTarget = nil
    end
    local dist, nearest = math.huge, nil
    for _, part in pairs(workspace.Live:GetChildren()) do
        if part:IsA("Model") and part:FindFirstChild("HumanoidRootPart") and part ~= char then
            local h = part:FindFirstChildOfClass("Humanoid")
            if h and h.Health > 0 then
                local d = (part.HumanoidRootPart.Position-hrp.Position).Magnitude
                if d < dist then dist=d nearest=part.HumanoidRootPart end
            end
        end
    end
    return nearest
end

local function stopHitbox()
    getgenv().ATCH = false
    for _, conn in pairs(getgenv().connections) do pcall(function() conn:Disconnect() end) end
    getgenv().connections = {}
    pcall(function() RunService:UnbindFromRenderStep("HBCam") end)
    pcall(function() camera.CameraSubject=humanoid camera.CameraType=Enum.CameraType.Custom end)
end

local function makeHeadInvisible()
    pcall(function()
        local head = char:FindFirstChild("Head")
        if head then
            head.LocalTransparencyModifier = 1
            -- Also hide face/hair accessories
            for _, v in ipairs(char:GetChildren()) do
                if v:IsA("Accessory") then
                    local mesh = v:FindFirstChildOfClass("SpecialMesh")
                    local part = v:FindFirstChildOfClass("Part") or v:FindFirstChild("Handle")
                    if part then part.LocalTransparencyModifier = 1 end
                end
            end
        end
    end)
end

local function startHitbox()
    stopHitbox()
    task.wait(0.1)
    getgenv().ATCH = true

    local e = Instance.new("Part")
    e.Size=Vector3.new(0.5,0.5,0.5) e.CanCollide=false e.CanTouch=false
    e.Anchored=true e.Transparency=1 e.Parent=char

    local p = Instance.new("Part",char)
    p.Size=Vector3.new(0.5,0.5,0.5) p.CanCollide=false p.CanTouch=false
    p.Anchored=true p.Transparency=1

    getgenv().val = e
    local WEIGHT = 15

    local function updateSubject()
        -- Keep head invisible every frame
        makeHeadInvisible()

        if cameraLocked then
            local en = findTarget()
            if en then
                local lookDir = Vector3.new(en.Position.X-hrp.Position.X,0,en.Position.Z-hrp.Position.Z)
                if lookDir.Magnitude > 0.1 then
                    hrp.CFrame = hrp.CFrame:Lerp(CFrame.lookAt(hrp.Position,hrp.Position+lookDir),0.15)
                end
            end
            camera.CameraType=Enum.CameraType.Custom
            camera.CameraSubject=humanoid
            return
        end
        if UserInputService.MouseBehavior==Enum.MouseBehavior.LockCenter then
            local lookXZ = Vector3.new(camera.CFrame.LookVector.X,0,camera.CFrame.LookVector.Z)
            if lookXZ.Magnitude > 0.01 then hrp.CFrame=CFrame.lookAt(hrp.Position,hrp.Position+lookXZ) end
        end
        if (camera.CFrame.Position-p.Position).Magnitude<1 or
           (camera.CFrame.Position-getgenv().val.Position).Magnitude<1 then
            camera.CameraSubject=humanoid p.Position=getgenv().val.Position
        else
            camera.CameraSubject=p
            TweenService:Create(p,TweenInfo.new(2/WEIGHT),{Position=e.Position}):Play()
        end
    end

    getgenv().connections["HBCam"] = RunService:BindToRenderStep("HBCam",Enum.RenderPriority.First.Value,updateSubject)
    getgenv().val = e

    getgenv().connections["HPWatch"] = RunService.Heartbeat:Connect(function()
        local hp = humanoid.Health
        if hp < lastHP then damagedUntil=tick()+0.4 end
        lastHP = hp
    end)

    getgenv().connections["HBMain"] = RunService.Heartbeat:Connect(function()
        if not active then return end
        local en = findTarget()
        if not en then return end
        getgenv().EN = en
        local name = tostring(en)
        updateHistory(name, en.Position)
        local vel,accel = getVelAndAccel(name)
        local xzSpeed = math.sqrt(vel.X*vel.X+vel.Z*vel.Z)
        local predTime = xzSpeed>40 and 1/10 or xzSpeed>25 and 1/16 or xzSpeed>10 and 1/24 or xzSpeed>3 and 1/35 or 0
        local predX = en.Position.X+vel.X*predTime+0.5*accel.X*predTime*predTime
        local predZ = en.Position.Z+vel.Z*predTime+0.5*accel.Z*predTime*predTime
        local dX,dZ = predX-en.Position.X, predZ-en.Position.Z
        local dMag = math.sqrt(dX*dX+dZ*dZ)
        if dMag > 7 then local s=7/dMag predX=en.Position.X+dX*s predZ=en.Position.Z+dZ*s end
        local old = hrp.CFrame
        local old2 = char.Head.CFrame
        e.CFrame = old2
        if not isSafeToRestore(old) then return end
        hrp.CFrame = CFrame.new(predX,441-9,predZ)*CFrame.Angles(math.rad(90),0,0)
        e.CFrame = old2
        RunService.RenderStepped:Wait()
        hrp.CFrame = old
        hrp.CFrame = CFrame.lookAt(hrp.Position,Vector3.new(en.Position.X,hrp.Position.Y,en.Position.Z))
    end)

    task.spawn(function()
        repeat task.wait(0.1) until not active or humanoid.Health<=0
        pcall(function() e:Destroy() end)
        pcall(function() p:Destroy() end)
        stopHitbox()
    end)
end

local function makeInvisible()
    if isInvis then return end
    isInvis = true
    originalSizes = {}
    pcall(function()
        for _, part in ipairs(char:GetDescendants()) do
            if part:IsA("BasePart") then
                originalSizes[part] = part.Size
                part.Size = Vector3.new(0.001,0.001,0.001)
            end
        end
    end)
end

local function makeVisible()
    if not isInvis then return end
    isInvis = false
    pcall(function()
        for part, size in pairs(originalSizes) do
            if part and part.Parent then part.Size = size end
        end
    end)
    originalSizes = {}
end

local _statusUpdateFn = nil

local function goSafe(reason)
    if safeCD then return end
    -- Check kill zone save limit
    if reason == "void" or reason == "edge" then
        killZoneSaves += 1
        if killZoneSaves > CFG.killZoneLimit then
            -- Stop saving — let the void kill us
            return
        end
    end
    safeCD = true
    isSafeMode = true
    if not safePlatform or not safePlatform.Parent then createPlatform() end
    if hrp then
        hrp.AssemblyLinearVelocity = Vector3.zero
        hrp.CFrame = CFrame.new(0, CFG.safeY+7, 0)
    end
    if _statusUpdateFn then _statusUpdateFn(reason, killZoneSaves) end
    task.delay(3, function()
        safeCD = false
        -- Don't exit safe mode here — HP system handles return
        if _statusUpdateFn then _statusUpdateFn("monitoring", killZoneSaves) end
    end)
end

-- ══════════════════════════════════════════
-- MAIN DETECTION LOOP
-- ══════════════════════════════════════════
RunService.Heartbeat:Connect(function()
    if not char or not hrp or not humanoid then return end
    if humanoid.Health <= 0 then return end

    local pos = hrp.Position
    local hpPct = (humanoid.Health/humanoid.MaxHealth)*100

    -- ── SAFE MODE HP LOGIC ──
    if isSafeMode then
        if hpPct >= CFG.safeReturnHP then
            -- Healthy enough — return to enemy
            local en = findTarget()
            if en and not safeCD then
                isSafeMode = false
                hrp.AssemblyLinearVelocity = Vector3.zero
                -- Teleport next to enemy
                local offset = (hrp.Position - en.Position).Unit
                if offset.Magnitude < 0.1 then offset = Vector3.new(1,0,0) end
                hrp.CFrame = CFrame.new(en.Position + Vector3.new(offset.X,0,offset.Z)*4 + Vector3.new(0,3,0))
                if _statusUpdateFn then _statusUpdateFn("returned", killZoneSaves) end
            end
        elseif hpPct < CFG.safeTriggerHP and not safeCD then
            -- Still low — keep us on platform
            if pos.Y < CFG.safeY - 5 then
                hrp.AssemblyLinearVelocity = Vector3.zero
                hrp.CFrame = CFrame.new(0, CFG.safeY+7, 0)
            end
        end
        return -- Don't run other checks while in safe mode
    end

    -- ── LOW HP — go safe ──
    if safeSettings.autoSafe and hpPct <= CFG.lowHPTrigger and not safeCD then
        goSafe("lowhp") return
    end

    -- ── VOID / EDGE ──
    if safeSettings.antiVoid then
        if pos.Y < 0 and not safeCD then goSafe("void") return end
        if (math.abs(pos.X) > CFG.edgeLimit or math.abs(pos.Z) > CFG.edgeLimit) and not safeCD then
            goSafe("edge") return
        end
    end

    -- ── PROXIMITY AUTO-SNAP ──
    if safeSettings.autoProximity and active and hpPct >= CFG.safeTriggerHP then
        local en = findTarget()
        if en then
            local dist = (en.Position - hrp.Position).Magnitude
            if dist > CFG.proximityRange then
                -- Too far — snap back to enemy
                hrp.AssemblyLinearVelocity = Vector3.zero
                local angle = math.random() * math.pi * 2
                local snapOffset = Vector3.new(math.cos(angle)*4, 3, math.sin(angle)*4)
                hrp.CFrame = CFrame.new(en.Position + snapOffset)
            end
        end
    end
end)

local function spawnDmgNum(amount, worldPos)
    local sp,onScreen = camera:WorldToViewportPoint(worldPos+Vector3.new(0,2.5,0))
    if not onScreen or sp.Z<=0 then return end
    local lbl = Instance.new("TextLabel")
    lbl.Size=UDim2.new(0,120,0,44)
    lbl.AnchorPoint=Vector2.new(0.5,0.5)
    lbl.Position=UDim2.new(0,sp.X,0,sp.Y)
    lbl.BackgroundTransparency=1
    lbl.Font=Enum.Font.GothamBold
    lbl.TextStrokeTransparency=0.1
    lbl.TextStrokeColor3=Color3.fromRGB(60,0,0)
    lbl.ZIndex=95
    local baseSize
    if amount >= 25 then
        lbl.Text="💥 -"..math.floor(amount)
        lbl.TextColor3=Color3.fromRGB(255,20,20) baseSize=34
    elseif amount >= 15 then
        lbl.Text="-"..math.floor(amount)
        lbl.TextColor3=Color3.fromRGB(255,100,20) baseSize=26
    elseif amount >= 8 then
        lbl.Text="-"..math.floor(amount)
        lbl.TextColor3=Color3.fromRGB(255,200,40) baseSize=20
    else
        lbl.Text="-"..math.floor(amount)
        lbl.TextColor3=Color3.fromRGB(240,240,240) baseSize=15
    end
    lbl.TextSize = baseSize * 0.5
    lbl.Parent = hitGui
    -- Pop in
    TweenService:Create(lbl,TweenInfo.new(0.12,Enum.EasingStyle.Back,Enum.EasingDirection.Out),{TextSize=baseSize}):Play()
    task.delay(0.12,function()
        TweenService:Create(lbl,TweenInfo.new(0.65,Enum.EasingStyle.Quint),{
            Position=UDim2.new(0,sp.X+math.random(-25,25),0,sp.Y-90),
            TextTransparency=1, TextStrokeTransparency=1,
        }):Play()
        task.delay(0.65,function() lbl:Destroy() end)
    end)
end

local function onHit(amount, worldPos)
    hitCombo += 1
    totalDmg += amount
    lastHitTime = tick()
    -- Screen flash
    vignette.BackgroundTransparency = 0.84
    TweenService:Create(vignette,TweenInfo.new(0.2),{BackgroundTransparency=1}):Play()
    spawnDmgNum(amount, worldPos)
    -- Combo update
    comboHolder.Visible = true
    comboNum.Text = tostring(hitCombo)
    local col = hitCombo>=15 and Color3.fromRGB(255,0,220)
        or hitCombo>=8 and Color3.fromRGB(255,80,0)
        or Color3.fromRGB(255,35,35)
    comboNum.TextColor3 = col
    -- Punch animation on number
    comboNum.TextSize = 58
    TweenService:Create(comboNum,TweenInfo.new(0.14,Enum.EasingStyle.Back),{TextSize=46}):Play()
end

RunService.Heartbeat:Connect(function()
    -- Combo decay
    if hitCombo > 0 and (tick()-lastHitTime) > 2.5 then
        hitCombo = 0
        TweenService:Create(comboHolder,TweenInfo.new(0.25),{}):Play()
        task.delay(0.25,function() comboHolder.Visible=false end)
    end
    -- Watch enemies
    if not active then return end
    for _, part in pairs(workspace.Live:GetChildren()) do
        if part:IsA("Model") and part ~= char then
            local h = part:FindFirstChildOfClass("Humanoid")
            local r = part:FindFirstChild("HumanoidRootPart")
            if h and r and h.Health > 0 then
                local id = tostring(part)
                local prev = lastEnemyHP[id] or h.Health
                local diff = prev - h.Health
                if diff > 0.5 and (r.Position-hrp.Position).Magnitude < 60 then
                    onHit(diff, r.Position)
                end
                lastEnemyHP[id] = h.Health
            end
        end
    end
end)

    for _,v in ipairs(plScroll:GetChildren()) do if v:IsA("TextButton") then v:Destroy() end end

        if p~=lp then

            b.TextXAlignment=Enum.TextXAlignment.Left b.Name=p.DisplayName b.Parent=plScroll
            Instance.new("UICorner",b).CornerRadius=UDim.new(0,5)

                for _,x in ipairs(plScroll:GetChildren()) do

                end
                if p.Character and p.Character:FindFirstChild("HumanoidRootPart") then

-- Live stat updates
RunService.Heartbeat:Connect(function()
    if not humanoid then return end
    local hpPct=(humanoid.Health/humanoid.MaxHealth)*100

    -- Pill knob color
    sDot.BackgroundColor3 = isSafeMode and C.orange
        or isInvis and C.purple
        or active and C.green
        or hpPct<=10 and C.red
        or C.dim

    if not active then return end

    local en = findTarget()
    if en then
        local dist=math.floor((en.Position-hrp.Position).Magnitude)
        distVal.Text=dist.." st"
        distVal.TextColor3=dist<10 and C.red or dist<25 and C.orange or C.blue

        local dn=en.Parent.Name
        for _,p in ipairs(game.Players:GetPlayers()) do if p.Character==en.Parent then dn=p.DisplayName break end end
        if not getgenv().lockedTarget then stTxt.Text="🎯 "..dn stTxt.TextColor3=C.green end

        for _,p in ipairs(game.Players:GetPlayers()) do
            if p~=lp and p.Character then
                local h=p.Character:FindFirstChildOfClass("Humanoid")
                if h then
                    local prev=lastHealth[p.Name] or h.Health
                    if h.Health<prev then
                        hitCount+=1 hitsVal.Text=tostring(hitCount)
                    end
                    lastHealth[p.Name]=h.Health
                end
            end
        end
    else
        distVal.Text="—" distVal.TextColor3=C.dim
        if not getgenv().lockedTarget then stTxt.Text="Scan" stTxt.TextColor3=C.dim end
    end

    dmgVal.Text=math.floor(totalDmg)
    safeVal.Text=isSafeMode and "ACTIVE" or "Standby"
    safeVal.TextColor3=isSafeMode and C.orange or C.dim
    savesVal.Text=(CFG.killZoneLimit-killZoneSaves).." left"
    savesVal.TextColor3=killZoneSaves>=CFG.killZoneLimit and C.red or C.orange
    proximVal.Text=safeSettings.autoProximity and "ON" or "OFF"
    proximVal.TextColor3=safeSettings.autoProximity and C.green or C.dim
end)
